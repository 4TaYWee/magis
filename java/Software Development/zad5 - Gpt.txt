import javafx.application.Application;
import javafx.stage.FileChooser;
import javafx.stage.Stage;
import javafx.scene.Scene;
import javafx.scene.chart.BarChart;
import javafx.scene.chart.CategoryAxis;
import javafx.scene.chart.NumberAxis;
import javafx.scene.chart.XYChart;
import javafx.scene.control.Button;
import javafx.scene.layout.VBox;
import javafx.scene.text.Text;
import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.io.IOException;

public class DataAnalyzer extends Application {
    
    private Text statusText;
    private BarChart<String, Number> barChart;

    @Override
    public void start(Stage primaryStage) {
        primaryStage.setTitle("Data Analyzer");

        statusText = new Text();

        Button openButton = new Button("Open CSV File");
        openButton.setOnAction(e -> openFile(primaryStage));

        CategoryAxis xAxis = new CategoryAxis();
        NumberAxis yAxis = new NumberAxis();
        barChart = new BarChart<>(xAxis, yAxis);
        barChart.setTitle("Data Analysis");

        VBox vbox = new VBox(openButton, statusText, barChart);
        Scene scene = new Scene(vbox, 800, 600);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    private void openFile(Stage primaryStage) {
        FileChooser fileChooser = new FileChooser();
        fileChooser.getExtensionFilters().add(new FileChooser.ExtensionFilter("CSV Files", "*.csv"));
        File selectedFile = fileChooser.showOpenDialog(primaryStage);

        if (selectedFile != null) {
            try {
                analyzeCSV(selectedFile);
            } catch (IOException e) {
                statusText.setText("Error: " + e.getMessage());
            }
        }
    }

    private void analyzeCSV(File file) throws IOException {
        // Clear previous data
        barChart.getData().clear();

        // Read CSV file and analyze data
        try (BufferedReader br = new BufferedReader(new FileReader(file))) {
            String line;
            while ((line = br.readLine()) != null) {
                // Parse CSV line and analyze data
                // Here you can implement your logic for data analysis

                // For demonstration purposes, let's assume we're analyzing numerical data
                String[] values = line.split(",");
                for (int i = 0; i < values.length; i++) {
                    String category = "Category " + (i + 1);
                    double value = Double.parseDouble(values[i]);

                    XYChart.Series<String, Number> series = new XYChart.Series<>();
                    series.getData().add(new XYChart.Data<>(category, value));
                    barChart.getData().add(series);
                }
            }
            statusText.setText("CSV file successfully analyzed.");
        }
    }

    public static void main(String[] args) {
        launch(args);
    }
}
