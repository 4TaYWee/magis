public class TaskController {
    private ObservableList<Task> tasks;
    private TableView<Task> taskTable;

    public TaskController() {
        tasks = new ObservableList<Task>();
        taskTable = new TableView<>(tasks);

        // Dodanie kolumn do tabeli
        TableColumn<Task, String> titleColumn = new TableColumn<>("Tytuł");
        titleColumn.setCellValueFactory(new PropertyValueFactory<>("title"));
        taskTable.getColumns().add(titleColumn);

        TableColumn<Task, String> descriptionColumn = new TableColumn<>("Opis");
        descriptionColumn.setCellValueFactory(new PropertyValueFactory<>("description"));
        taskTable.getColumns().add(descriptionColumn);

        TableColumn<Task, Boolean> doneColumn = new TableColumn<>("Wykonane");
        doneColumn.setCellValueFactory(new PropertyValueFactory<>("isDone"));
        doneColumn.setCellFactory(new CheckBoxTableCell<>(tasks));
        taskTable.getColumns().add(doneColumn);
    }
